const opts = {
  gridx: 10,
  gridy: 20,
  tickDuration: 1000,
  heatRoomTime: 5000,
}
 
class game
{
  constructor(opts) {
    this.idGame = '#G123wopafkl90123jker';
    this.players = [];
    this.spectators = [];
    this.gameHasStarted = false;
    this.gameStartIn = opt.heatRoomTime;
    this.tick = 0;
    setTimeout(this.launchGame, this.heatRoomTime)
  }
 
  broadcast_in_room(actionName, payload) {
    this.players.forEach((player, index) => {
      player.send(actionName, { payload });
    });
  }
 
  startGame() {
    this.tick = 0;
    this.player.forEach((player) => {
      player.resetForNextMatch(opts);
      player.on('action_valid', (action) => {
        this.broadcast_in_room('action', action)
      });
    });
    this.loop = setInterval(this.nextTick, tickDuration);
  }
 
  stopGame() {
    clearInterval(this.loop);
    this.loop = undefined;
  }
 
  nextTick() {
    this.player.forEach((player, index) => {
      player.sendNextTick(this.tick);
    });
    this.tick = this.tick + 1;
 
    this.broadcast_in_room('buffers', this.concatAllBuffers());
  }
 
  concatAllBuffers() {
    let totalLength = 0;
    let buffersToSend = this.players.map((player, index) => {
      totalLength += player.buffer.length;
      return player.buffer;
    });
    return Buffer.concat(buffersToSend, totalLength);
  }
 
  nextMatch() {
    this.stopGame();
    this.startGame();
  }
 
  addPlayer(player) {
    this.players.push(player);
  }
 
  removePlayer(player) {
    player.destroy();
 
    //  this.players = {}
  }
 
}
 
export default game;
